CREATE DATABASE videoMonitoramento;
USE videoMonitoramento;

SET time_zone='America/Sao_Paulo';

CREATE TABLE IF NOT EXISTS `Server` (
  Id CHAR(36) CHARSET utf8mb4,
  Name LONGTEXT CHARSET utf8mb4,
  Ip LONGTEXT CHARSET utf8mb4,
  Port INT,
  PRIMARY KEY (`Id`)
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS `Video` (
 Id CHAR(36) CHARSET utf8mb4 PRIMARY KEY,
 Description LONGTEXT CHARSET utf8mb4,
 SizeInBytes INT,
 Date TIMESTAMP NOT NULL,
 ServerId CHAR(36) CHARSET utf8mb4,
 CONSTRAINT fk_server_id FOREIGN KEY (ServerId)
 REFERENCES Server(Id)
 ON DELETE CASCADE
 ON UPDATE CASCADE
) ENGINE=InnoDB;
